/* eslint-disable */
import { useEffect, useState, useRef } from "react";
import { motion, useAnimation } from 'framer-motion';
import { ArrowDown } from "lucide-react";

export const HeroSection = () => {
  const [isDarkMode, setIsDarkMode] = useState(false);
  const [animationProgress, setAnimationProgress] = useState(0);
  const [isAnimationComplete, setIsAnimationComplete] = useState(false);
  const containerRef = useRef(null);
  const scrollAccumulator = useRef(0);

  // Detect dark mode
  useEffect(() => {
    const checkTheme = () => {
      const isDark = document.documentElement.classList.contains('dark');
      setIsDarkMode(isDark);
    };
    
    checkTheme();
    const observer = new MutationObserver(checkTheme);
    observer.observe(document.documentElement, { attributes: true, attributeFilter: ['class'] });
    
    return () => observer.disconnect();
  }, []);

  // Handle scroll-based zoom animation
  useEffect(() => {
    if (isAnimationComplete) return;

    const handleWheel = (e) => {
      e.preventDefault();
      
      // Accumulate scroll delta
      scrollAccumulator.current += e.deltaY;
      
      // Convert to progress (0 to 1)
      // Adjust 2000 to control how much scrolling is needed
      const progress = Math.min(Math.max(scrollAccumulator.current / 2000, 0), 1);
      setAnimationProgress(progress);
      
      // Complete animation at 100%
      if (progress >= 1) {
        setIsAnimationComplete(true);
      }
    };

    window.addEventListener('wheel', handleWheel, { passive: false });
    
    return () => {
      window.removeEventListener('wheel', handleWheel);
    };
  }, [isAnimationComplete]);

  // Lock/unlock body scroll
  useEffect(() => {
    if (!isAnimationComplete) {
      document.body.style.overflow = 'hidden';
      document.body.style.height = '100vh';
    } else {
      document.body.style.overflow = 'auto';
      document.body.style.height = 'auto';
    }

    return () => {
      document.body.style.overflow = 'auto';
      document.body.style.height = 'auto';
    };
  }, [isAnimationComplete]);

  // Animation values based on progress (smoother transitions)
  const scale = 1 + (animationProgress * 149); // 1 to 150
  const textOpacity = animationProgress < 0.7 ? 1 : Math.max(0, 1 - ((animationProgress - 0.7) / 0.3));
  const overlayOpacity = animationProgress < 0.75 ? 0 : Math.min(1, (animationProgress - 0.75) / 0.2);
  const aboutContentOpacity = animationProgress < 0.85 ? 0 : Math.min(1, (animationProgress - 0.85) / 0.15);
  const aboutContentY = Math.max(0, 50 - (animationProgress * 60));

  return (
    <>
      {/* ZOOM ANIMATION VIEWPORT */}
      <div 
        ref={containerRef} 
        className={`fixed inset-0 w-full h-screen overflow-hidden flex items-center justify-center transition-colors duration-500 ${
          isDarkMode ? 'bg-black' : 'bg-gray-50'
        } ${isAnimationComplete ? 'pointer-events-none' : ''}`}
        style={{ 
          zIndex: isAnimationComplete ? -1 : 0,
          opacity: isAnimationComplete ? 0 : 1,
          transition: 'opacity 0.8s ease-out, z-index 0s 0.8s'
        }}
      >
        {/* LAYER 1: The Zooming Name */}
        <motion.div 
          style={{ 
            scale,
            opacity: textOpacity
          }}
          className="absolute z-10 origin-center"
        >
          <h1 className={`text-[20vw] font-black leading-none tracking-tighter select-none flex ${
            isDarkMode ? 'text-white' : 'text-black'
          }`}>
            <span>VA</span>
            <span className="relative">R</span>
            <span>UN</span>
          </h1>
        </motion.div>

        {/* LAYER 2: The "Inside the Letter" Background */}
        <motion.div
          style={{ opacity: overlayOpacity }}
          className={`absolute inset-0 z-20 pointer-events-none ${
            isDarkMode ? 'bg-white' : 'bg-black'
          }`}
        />

        {/* LAYER 3: The About Content */}
        <motion.div 
          style={{ 
            opacity: aboutContentOpacity,
            y: aboutContentY
          }}
          transition={{ duration: 0.3, ease: "easeOut" }}
          className="absolute z-30 inset-0 flex items-center justify-center"
        >
          <div className="max-w-4xl px-6 text-center">
            <h2 className={`text-4xl md:text-6xl font-bold mb-6 ${
              isDarkMode ? 'text-black' : 'text-white'
            }`}>
              About Me
            </h2>
            <p className={`text-lg md:text-2xl leading-relaxed max-w-2xl mx-auto ${
              isDarkMode ? 'text-gray-800' : 'text-gray-200'
            }`}>
              I am a passionate developer who builds immersive web experiences. 
              You have just scrolled <em>inside</em> my name to find this content.
            </p>
          </div>
        </motion.div>

        {/* Scroll Indicator */}
        <motion.div 
          style={{ 
            opacity: animationProgress < 0.2 ? 1 - (animationProgress / 0.2) : 0
          }}
          className="absolute bottom-10 left-1/2 -translate-x-1/2 z-10 flex flex-col items-center gap-2"
        >
          <span className={`text-sm font-medium ${isDarkMode ? 'text-white/50' : 'text-black/50'}`}>
            Scroll to enter
          </span>
          <ArrowDown className={`w-5 h-5 animate-bounce ${isDarkMode ? 'text-white' : 'text-black'}`} />
        </motion.div>
      </div>

      {/* MAIN CONTENT - Visible after animation */}
      <div 
        className="relative"
        style={{ 
          opacity: isAnimationComplete ? 1 : 0,
          transition: 'opacity 0.8s ease-in',
          pointerEvents: isAnimationComplete ? 'auto' : 'none'
        }}
      >
        {/* About Section (appears after zoom) */}
        <div className={`min-h-screen flex items-center justify-center ${
          isDarkMode ? 'bg-white' : 'bg-black'
        }`}>
          <div className="max-w-4xl px-6 text-center py-24">
            <h2 className={`text-4xl md:text-6xl font-bold mb-6 ${
              isDarkMode ? 'text-black' : 'text-white'
            }`}>
              About Me
            </h2>
            <p className={`text-lg md:text-2xl leading-relaxed max-w-2xl mx-auto ${
              isDarkMode ? 'text-gray-800' : 'text-gray-200'
            }`}>
              I am a passionate developer who builds immersive web experiences. 
              You just scrolled <em>inside</em> my name to discover this page.
            </p>
          </div>
        </div>

        {/* Skills Section */}
        <div className={`relative py-24 ${
          isDarkMode ? 'bg-gray-900' : 'bg-white'
        }`}>
          <div className="max-w-6xl mx-auto px-6">
            <h2 className={`text-3xl font-bold mb-12 text-center ${isDarkMode ? 'text-white' : 'text-black'}`}>
              Technical Arsenal
            </h2>
            
            <div className="grid grid-cols-2 md:grid-cols-4 gap-4">
              {['React', 'Next.js', 'TypeScript', 'Node.js', 'Tailwind', 'Motion', 'Postgres', 'Docker'].map((skill) => (
                <div key={skill} className={`p-6 rounded-xl border ${
                  isDarkMode 
                    ? 'bg-gray-800 border-gray-700 text-white' 
                    : 'bg-gray-50 border-gray-200 text-gray-900'
                }`}>
                  <p className="font-semibold text-center">{skill}</p>
                </div>
              ))}
            </div>
          </div>
        </div>
      </div>
    </>
  );
};
